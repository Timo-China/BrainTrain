157

160  // current vertex coordinates

163  //normal coordinates are calculated on the fly
164

167  alpha = 0;
168  double curRadius = curFunc->func(z);

172  curRadius * sin(alpha),//Y
173  z)); //Z
174  normals->push_back(Vec3d(
175  curRadius * cos(alpha), //X
176  curRadius * sin(alpha), //Y
177  -curRadius * curFunc->derivation(z))); //Z
178  alpha -= alphastep;
179  }
180  z += zstep;
181  if (curFunc->end < z - 1e-8)
182  curFunc = curFunc->nextFunc;
183  }
184 
185  //set strip connections
186  for (int i = 0; i < hsteps; i++) {
187 
188  for (int j = 0; j <= rsteps; j++) {
189  indices->push_back(i*(rsteps + 1) + j);
190  indices->push_back((i + 1)*(rsteps + 1) + j);
191  }
192  indices->push_back((i + 1)*(rsteps + 1) + rsteps);
193  indices->push_back((i + 1)*(rsteps + 1));
194  }
195 
196  //normalize the normals
197  for (auto cnt = 0; cnt < normals->size(); cnt++) {
198  normals->at(cnt).normalize();
199  }
200 
201  rect->setVertexArray(vertices);
202  rect->addPrimitiveSet(indices);
203  rect->setNormalArray(normals);
